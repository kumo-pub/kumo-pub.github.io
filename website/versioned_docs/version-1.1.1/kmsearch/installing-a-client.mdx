# 安装客户端

import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

我们提供以下客户端库：

- [JavaScript](https://gitee.com/kumo-pub/kumosdk-js)
- [Python](https://gitee.com/kumo-pub/kumosdk-py)
- [Go](https://gitee.com/kumo-pub/kumosdk-go)

<Tabs>
    <TabItem value="Js-shell" label="Js-shell">

        ```js

        <script src="https://cdn.jsdelivr.net/npm/kumosdk@0/dist/kumosdk.min.js"></script>

        ```
    </TabItem>
    <TabItem value="JavaScript" label="JavaScript">

        ```shell
        // npm install kumosdk @babel/runtime
        npm install kumosdk @babel/runtime
        ```

    </TabItem>
    <TabItem value="go" label="go">

        ```shell

        go get gitee.com/kumo-pub/kumosdk-go/kumosearch

        ```
    </TabItem>
    <TabItem value="python" label="python">

        ```shell

        pip install kumosdk

        ```
    </TabItem>
</Tabs>
